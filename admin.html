Auto-Sync Orders to Google Sheets
📊 Once connected, every single order from your website automatically appears in a Google Sheet — perfect for GST filing and sharing with your accountant. Completely free!
Step 1 — Create a Google Sheet

Go to sheets.google.com → Create new sheet → Name it "Shivar Orders"

Step 2 — Create an Apps Script

In your sheet: click Extensions → Apps Script → delete all existing code → paste this:

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        'Order Ref','Date & Time','Status',
        'Customer Name','Phone','Delivery Address','PIN Code',
        'Items Ordered','Product Total (₹)','GST 5% (₹)','HSN Code',
        'Shipping Zone','Shipping (₹)','Grand Total (₹)',
        'Payment Method','Screenshot Uploaded'
      ]);
      var hdr = sheet.getRange(1,1,1,16);
      hdr.setFontWeight('bold')
         .setBackground('#0a1f0d')
         .setFontColor('#ffffff');
      sheet.setFrozenRows(1);
    }
    var d = JSON.parse(e.postData.contents);
    var itemStr = d.items.map(function(i) {
      return i.weight + ' x' + i.qty + ' = ₹' + i.total;
    }).join(' | ');
    sheet.appendRow([
      d.ref, d.dateLocal, d.status,
      d.customer.name, d.customer.phone,
      d.customer.address, d.customer.pincode,
      itemStr,
      d.pricing.productTotal, d.pricing.gst, d.pricing.hsnCode,
      d.pricing.shippingZone, d.pricing.shipping, d.pricing.grandTotal,
      d.payment.method,
      d.payment.screenshotUploaded ? 'Yes' : 'No'
    ]);
    var row = sheet.getLastRow();
    sheet.getRange(row,1,1,16).setBackground('#f0faf4');
    return ContentService
      .createTextOutput(JSON.stringify({success:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
Step 3 — Deploy as Web App

Click Deploy → New deployment → Web app
Execute as: Me · Who has access: Anyone
Click Deploy → Authorize → Copy the Web App URL

Step 4 — Paste URL into index.html

In GitHub, open index.html → find this line near the bottom → paste your URL:

var SHEET_URL = 'PASTE_YOUR_APPS_SCRIPT_URL_HERE';
✅ Setup takes about 10 minutes. Completely free. Works on any device. Every new order appears in your sheet within seconds.
